def getIntersectionPoints(meanS, meanN, stdS, stdN):
    intersectionPoints = np.roots([1 / (2 * stdS**2) - 1 / (2 * stdN**2), meanN / (stdN**2) - meanS / (stdS**2), meanS**2 / (2 * stdS**2) - meanN**2 / (2 * stdN**2) - np.log(stdN / stdS)])
    return intersectionPoints

sexInterPoints = getIntersectionPoints(dataBasedOnSurvived.loc['Sex']['Mean Of Survived'], dataBasedOnSurvived.loc['Sex']['Mean Of Not Survived'], dataBasedOnSurvived.loc['Sex']['Std Of Survived'], dataBasedOnSurvived.loc['Sex']['Std Of Not Survived'])

def doesSurvive(normalizedtest, intersectionPoints):
    if(intersectionPoints[0] < intersectionPoints[1]):
        min = intersectionPoints[0]
        max = intersectionPoints[1]
    else:
        min = intersectionPoints[1]
        max = intersectionPoints[0]

    if (min < normalizedtest < max):
        return 0
    return 1
